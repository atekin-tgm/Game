[docs]    def newGame(self):
        """
        the stats are being resetted
        """
        self.model.spiele += 1
        self.model.übrig = 15
        self.model.korrekt = 0
        self.model.falsch = 0
        self.model.letztes = 1

        for b in self.buttons:
            b.setEnabled(True)

        self.upd()
        self.buttonText()



[docs]    def checkButton(self, button):
        """
        the stats are being set
        :param button: pressed button
        """
        if int(button.text()) == self.model.last_one:
            button.setEnabled(False)
            self.model.letztes += 1
            self.model.übrig -= 1
            self.model.korrekt += 1
        else:
            self.model.falsch += 1

        self.upd()
